cmake_minimum_required(VERSION 2.8.3)

  rospy
  std_msgs
  message_generation

)

  rospy
  std_msgs
  message_runtime
  i_believe_i_can_fly_common
  i_believe_i_can_fly_person_detection
)

include_directories(
  ${catkin_INCLUDE_DIRS}
